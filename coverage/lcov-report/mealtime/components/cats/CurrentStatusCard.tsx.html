import * as React from "react";
import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card";
import { Progress } from "@/components/ui/progress";
import { H3, P, Small } from "@/components/ui/typography";
import { Info } from "lucide-react";
 
interface CurrentStatusCardProps {
  currentWeight: number;
  targetWeight: number;
  healthTip: string;
}
 
export function CurrentStatusCard({ currentWeight, targetWeight, healthTip }: CurrentStatusCardProps) {
  const progress = Math.min((currentWeight / targetWeight) * 100, 100);
  return (
    <Card className="w-full max-w-md">
      <CardHeader>
        <CardTitle>Current Weight</CardTitle>
        <CardDescription>{currentWeight} kg</CardDescription>
      </CardHeader>
      <CardContent>
        <div className="flex items-center justify-between mb-2">
          <Small>Target Weight</Small>
          <span>{targetWeight} kg</span>
        </div>
        <Progress value={progress} aria-label="Weight progress" />
        <div className="mt-4 flex items-start gap-2 p-3 rounded-md bg-accent text-accent-foreground">
          <Info className="w-5 h-5 text-primary mt-0.5" aria-hidden />
          <P className="m-0 text-sm">{healthTip}</P>
        </div>
      </CardContent>
    </Card>
  );
}
